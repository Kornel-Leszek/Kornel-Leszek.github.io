<input type="button" value="Oblicz" onclick="fun_les()">
<script>
function fun_les() {
    var opcja_dana_les = prompt("Podaj kolor", "");
    opcja_dana_les = opcja_dana_les.toLowerCase();

    if (opcja_dana_les == "1" || opcja_dana_les == "czerwony") {
        document.write('<font color="red" size="7">wczytałeś kolor czerwony</font>');
    }
    else if (opcja_dana_les == "2" || opcja_dana_les == "niebieski") {
        document.write('<font color="blue" size="1">wczytałeś kolor niebieski</font>');
    }
}
</script>
