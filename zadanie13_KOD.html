<!-- Kod z pliku zadanie13_Lesz.html --> <script> // Wczytanie liczby m var input = prompt("Podaj liczbę naturalną m (m > 0):", "4"); // sprawdzenie czy podano wartość if (input === null) { document.write("<p>Przerwano - nie podano wartości.</p>"); } else { var m = Number(input); if (!Number.isFinite(m) || m <= 0 || Math.floor(m) !== m) { document.write("<p style='color:red;'>Błąd: proszę podać poprawną liczbę naturalną większą od 0.</p>"); } else { var a = m + 1; var b = m * (m + 1); document.write("<p>Wybrano m = <strong>" + m + "</strong></p>"); document.write("<p class='frac'>Zatem: <span style='font-size:1.2em;'>1/" + m + " = 1/" + a + " + 1/" + b + "</span></p>"); var left = 1 / m; var right = 1 / a + 1 / b; document.write("<div class='example'>"); document.write("<p>Wartości dziesiętnie: 1/" + m + " = " + left.toFixed(10) + "</p>"); document.write("<p>1/" + a + " + 1/" + b + " = " + right.toFixed(10) + "</p>"); if (Math.abs(left - right) < 1e-12) { document.write("<p style='color:green;'><strong>Sprawdzenie: równość zachowana.</strong></p>"); } else { document.write("<p style='color:orange;'><strong>Uwaga:</strong> suma różni się minimalnie ze względu na arytmetykę zmiennoprzecinkową.</p>"); } document.write("</div>"); } } </script>
