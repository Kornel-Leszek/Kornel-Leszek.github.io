Leszek - Zadanie 22 - Kod

function Pole_leszek(a_les, b_les) {
  return a_les * b_les;
}

function obwod_leszek(a_les, b_les) {
  return 2 * (a_les + b_les);
}

function oblicz_les() {
  var a_les = parseFloat(document.getElementById("a_dana_les").value);
  var b_les = parseFloat(document.getElementById("b_dana_les").value);

  if (isNaN(a_les) || isNaN(b_les)) {
    alert("Podaj liczby!");
    return;
  }

  document.getElementById("wyprowadz_pole_les").value = Pole_leszek(a_les, b_les);
  document.getElementById("wyprowadz_obwod_les").value = obwod_leszek(a_les, b_les);
}
