var a = parseFloat(prompt("Podaj a:", ""));
var b = parseFloat(prompt("Podaj b:", ""));

function suma_les(a, b) {
    return a + b;
}

function odejmowanie_les(a, b) {
    return a - b;
}

function mnozenie_les(a, b) {
    return a * b;
}

function od_kwadratu_les(a, b) {
    let a2 = a ** 2;
    let b2 = b ** 2;
    document.write("a² = " + a2 + "<br>");
    document.write("b² = " + b2 + "<br>");
}

function reszta_z_dzielenia_a_przez_b_les(a, b) {
    return a % b;
}

document.write("suma = " + suma_les(a, b) + "<br>");
document.write("różnica = " + odejmowanie_les(a, b) + "<br>");
document.write("iloczyn = " + mnozenie_les(a, b) + "<br>");
od_kwadratu_les(a, b);
document.write("reszta z dzielenia a przez b = " + reszta_z_dzielenia_a_przez_b_les(a, b) + "<br>");
