function AB_les() {
    var xa_les = parseFloat(document.getElementById("xa").value);
    var ya_les = parseFloat(document.getElementById("ya").value);
    var xb_les = parseFloat(document.getElementById("xb").value);
    var yb_les = parseFloat(document.getElementById("yb").value);

    var ab_les = Math.sqrt(Math.pow(xa_les - xb_les, 2) + Math.pow(ya_les - yb_les, 2));
    document.getElementById("wyniki").innerHTML = 
        "Długość boku AB = " + ab_les.toFixed(2);
}
